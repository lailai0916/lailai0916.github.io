:::info[[洛谷 P3382 三分](https://www.luogu.com.cn/problem/P3382)]

给出一个 $N$ 次函数，保证在范围 $[l, r]$ 内存在一点 $x$，使得 $[l, x]$ 上单调增，$[x, r]$ 上单调减。试求出 $x$ 的值。

:::

```cpp
#include <bits/stdc++.h>
using namespace std;

const double eps=1e-6;
const int N=15;
double a[N];
int n;
double f(double x)
{
	double res=0;
	for(int i=n;i>=0;i--)res=res*x+a[i];
	return res;
}
int main()
{
	cin.tie(nullptr);
	ios::sync_with_stdio(false);
	double l,r;
	cin>>n>>l>>r;
	for(int i=n;i>=0;i--)
	{
		cin>>a[i];
	}
	while(r-l>eps)
	{
		double lmid=(l*2+r)/3;
		double rmid=(r*2+l)/3;
		if(f(lmid)>f(rmid))r=rmid;
		else l=lmid;
	}
	cout<<fixed<<setprecision(6)<<l<<'\n';
	return 0;
}
```
