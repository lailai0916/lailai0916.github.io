# 对拍器

```bat
@echo off
setlocal enabledelayedexpansion
set num=0

:loop
	set /a num+=1
	gen.exe > in
	std.exe < in > ans
	sol.exe < in > out
	fc out ans > nul 2>&1
	if errorlevel 1 (
		powershell -command "Write-Host '>>> Error #!num!' -ForegroundColor Red"
		fc out ans
		goto end
	) else (
		powershell -command "Write-Host '>>> Success #!num!' -ForegroundColor Green"
		goto loop
	)

:end
pause
```
