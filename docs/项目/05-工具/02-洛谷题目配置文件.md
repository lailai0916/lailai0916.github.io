## main.cpp

```cpp
#include <bits/stdc++.h>
using namespace std;

const string name="";
const int len=2;
int id=0;
void subtask(int n,int s)
{
	for(int i=1;i<=n;i++)
	{
		cout<<name<<setfill('0')<<id<<setw(len)<<i<<".in:"<<'\n';
		cout<<"  subtaskId: "<<id<<'\n';
		cout<<"  score: "<<s<<'\n';
		cout<<'\n';
	}
	id++;
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	
	return 0;
}
```
