## main.cpp

```cpp
#include <bits/stdc++.h>
using namespace std;

const string name="";
const int timeLimit=1500;
const int memoryLimit=256000;
int subtaskId=0;
void subtask(int num,int score)
{
	for(int i=1;i<=num;i++)
	{
		cout<<name<<subtaskId<<setfill('0')<<setw(2)<<i<<".in:"<<'\n';
		cout<<"  timeLimit: "<<timeLimit<<'\n';
		cout<<"  memoryLimit: "<<memoryLimit<<'\n';
		cout<<"  score: "<<score<<'\n';
		cout<<"  subtaskId: "<<subtaskId<<'\n';
		cout<<'\n';
	}
	subtaskId++;
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(nullptr);
	
	return 0;
}
```
